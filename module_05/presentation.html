1. INSERT — Добавление данных

Базовый синтаксис

-- Синтаксис INSERT
INSERT INTO таблица (столбец1, столбец2, ...)
VALUES (значение1, значение2, ...);
Правило: Порядок значений должен соответствовать порядку указанных столбцов. Столбцы с DEFAULT или SERIAL можно не указывать.

INSERT: Одна строка

Добавляем новую причину простоя

INSERT INTO dim_downtime_reason (
    reason_name, reason_code, category, description
)
VALUES (
    'Замена конвейерной ленты',
    'CONV_BELT',
    'плановый',
    'Плановая замена изношенной конвейерной ленты'
);
SERIAL: Столбец reason_id заполнится автоматически из последовательности.

INSERT: Несколько строк

Добавляем новые сорта руды

INSERT INTO dim_ore_grade (
    grade_name, grade_code, fe_content_min, fe_content_max, description
)
VALUES
    ('Премиум',     'PREM', 65.00, 72.00, 'Руда высшего качества'),
    ('Стандарт',    'STD',  55.00, 64.99, 'Руда стандартного качества'),
    ('Низкосортная','LOW',  40.00, 54.99, 'Руда пониженного качества');
Преимущество: Один оператор вместо трёх — меньше накладных расходов на транзакции и сетевые обращения.

Автогенерация ключей

SERIAL vs GENERATED ALWAYS AS IDENTITY

SERIAL (классический)

CREATE TABLE example_serial (
    id   SERIAL PRIMARY KEY,
    name VARCHAR(100)
);

-- Можно вставить вручную:
INSERT INTO example_serial (id, name)
VALUES (999, 'Ручной ID');

IDENTITY (современный)

CREATE TABLE example_identity (
    id   INT GENERATED ALWAYS
         AS IDENTITY PRIMARY KEY,
    name VARCHAR(100)
);

-- Ручная вставка запрещена!
-- Но можно через OVERRIDING:
INSERT INTO example_identity
    (id, name)
OVERRIDING SYSTEM VALUE
VALUES (999, 'Ручной ID');
Рекомендация PostgreSQL 10+: Используйте GENERATED ALWAYS AS IDENTITY вместо SERIAL для новых проектов.

Что происходит «под капотом»

SERIAL — синтаксический сахар

-- Когда вы пишете:
CREATE TABLE example (id SERIAL PRIMARY KEY);

-- PostgreSQL на самом деле выполняет:
CREATE SEQUENCE example_id_seq;
CREATE TABLE example (
    id INTEGER NOT NULL DEFAULT nextval('example_id_seq')
);
ALTER SEQUENCE example_id_seq OWNED BY example.id;
⚠️ Проблемы SERIAL:
  • Последовательность — отдельный объект, не привязанный жёстко к столбцу
  • При DROP COLUMN последовательность не удаляется автоматически → мусор в БД
  • Права на последовательность нужно выдавать отдельно
  • Нет защиты от ручной вставки — можно сломать уникальность

Два варианта IDENTITY

GENERATED ALWAYS vs GENERATED BY DEFAULT

GENERATED ALWAYS

id INT GENERATED ALWAYS
       AS IDENTITY PRIMARY KEY

-- Ручная вставка → ОШИБКА:
INSERT INTO t (id, name)
VALUES (1, 'test');
-- ERROR: cannot insert
-- a non-DEFAULT value

-- Обход (явный):
INSERT INTO t (id, name)
OVERRIDING SYSTEM VALUE
VALUES (1, 'test');

GENERATED BY DEFAULT

id INT GENERATED BY DEFAULT
       AS IDENTITY PRIMARY KEY

-- Ручная вставка разрешена:
INSERT INTO t (id, name)
VALUES (1, 'test');
-- OK!

-- Работает как SERIAL,
-- но по стандарту SQL
-- и с правильным управлением
-- последовательностью
💡 Когда что использовать: GENERATED ALWAYS — для суррогатных ключей (защита от ошибок). GENERATED BY DEFAULT — когда нужна миграция данных с существующими ID.

🤔 Вопрос для обсуждения

SERIAL vs IDENTITY в контексте «Руда+»

В системе MES предприятия «Руда+» таблица dim_equipment использует SERIAL для генерации ID оборудования.

При миграции данных из старой системы оператор вручную вставил ID, совпавший с будущим значением последовательности.

Что произойдёт при следующей автоматической вставке?

Ответ: Произойдёт ошибка duplicate key value violates unique constraint, потому что SERIAL не знает о ручных вставках. С GENERATED ALWAYS AS IDENTITY такая ситуация невозможна — ручная вставка была бы заблокирована изначально.

INSERT: Значения по умолчанию

DEFAULT и DEFAULT VALUES

-- Использование DEFAULT для отдельных столбцов
INSERT INTO dim_equipment (
    equipment_type_id, mine_id, equipment_name,
    inventory_number, status
)
VALUES (
    1, 1, 'ПДМ-19', 'INV-LHD-019', DEFAULT  -- status = 'active'
);

-- Все столбцы заполняются значениями по умолчанию
-- (полезно для таблиц-логов с timestamp)
INSERT INTO dim_date DEFAULT VALUES; -- ошибка, если нет DEFAULT у NOT NULL
Когда использовать DEFAULT: для столбцов со значениями по умолчанию (status, created_at, loaded_at и др.)

INSERT ... SELECT

Добавление данных из запроса

-- Скопировать записи о добыче из временной staging-таблицы
-- в основную факт-таблицу
INSERT INTO fact_production (
    date_id, shift_id, mine_id, shaft_id,
    equipment_id, operator_id, location_id, ore_grade_id,
    tons_mined, tons_transported, trips_count,
    distance_km, fuel_consumed_l, operating_hours
)
SELECT
    s.date_id, s.shift_id, s.mine_id, s.shaft_id,
    s.equipment_id, s.operator_id, s.location_id, s.ore_grade_id,
    s.tons_mined, s.tons_transported, s.trips_count,
    s.distance_km, s.fuel_consumed_l, s.operating_hours
FROM staging_production s
WHERE s.is_validated = TRUE
  AND NOT EXISTS (
      SELECT 1 FROM fact_production fp
      WHERE fp.date_id = s.date_id
        AND fp.shift_id = s.shift_id
        AND fp.equipment_id = s.equipment_id
        AND fp.operator_id = s.operator_id
  );
ETL-паттерн: INSERT...SELECT — основной способ перемещения данных между staging и аналитическими таблицами.

Получение вставленных данных

2. UPDATE — Изменение данных

Базовый синтаксис

UPDATE таблица
SET столбец1 = значение1,
    столбец2 = значение2
WHERE условие;
ВНИМАНИЕ: UPDATE без WHERE обновит ВСЕ строки таблицы! Всегда проверяйте условие WHERE перед выполнением.

UPDATE: Базовый пример

Обновление статуса оборудования

-- Перевести оборудование на техобслуживание
UPDATE dim_equipment
SET status = 'maintenance'
WHERE equipment_id = 5;

-- Обновить несколько столбцов
UPDATE dim_equipment
SET status = 'active',
    has_navigation = TRUE,
    has_video_recorder = TRUE
WHERE inventory_number = 'INV-LHD-005';
Совет: Перед UPDATE выполните SELECT с тем же WHERE, чтобы убедиться, какие строки будут затронуты.

UPDATE с подзапросом

Массовое обновление на основе связанных данных

-- Перевести в 'maintenance' всё оборудование,
-- у которого были внеплановые простои > 120 минут за последнюю неделю
UPDATE dim_equipment
SET status = 'maintenance'
WHERE equipment_id IN (
    SELECT DISTINCT d.equipment_id
    FROM fact_equipment_downtime d
    JOIN dim_date dd ON d.date_id = dd.date_id
    WHERE d.is_planned = FALSE
      AND d.duration_min > 120
      AND dd.full_date >= CURRENT_DATE - INTERVAL '7 days'
);
Бизнес-логика: Если самосвал простоял внепланово более 2 часов — его нужно отправить на диагностику.

UPDATE ... FROM

Расширение PostgreSQL для UPDATE с JOIN

-- Обновить сорт руды в записях о добыче
-- на основе результатов лабораторного анализа
UPDATE fact_production fp
SET ore_grade_id = oq.ore_grade_id
FROM fact_ore_quality oq
WHERE fp.date_id = oq.date_id
  AND fp.shift_id = oq.shift_id
  AND fp.shaft_id = oq.shaft_id
  AND fp.ore_grade_id IS NULL
  AND oq.ore_grade_id IS NOT NULL;
UPDATE ... FROM — специфика PostgreSQL. В стандартном SQL используют коррелированные подзапросы. 
-- Стандартный SQL (также работает в PostgreSQL)
UPDATE fact_production fp
SET ore_grade_id = (
    SELECT oq.ore_grade_id
    FROM fact_ore_quality oq
    WHERE fp.date_id = oq.date_id AND fp.shift_id = oq.shift_id
      AND fp.shaft_id = oq.shaft_id AND oq.ore_grade_id IS NOT NULL
    LIMIT 1
)
WHERE fp.ore_grade_id IS NULL;

UPDATE ... RETURNING

Получение изменённых данных

-- Перевести все ПДМ шахты «Северная» в статус обслуживания
-- и получить список затронутого оборудования
UPDATE dim_equipment e
SET status = 'maintenance'
FROM dim_equipment_type et
WHERE e.equipment_type_id = et.equipment_type_id
  AND et.type_code = 'LHD'
  AND e.mine_id = 1
  AND e.status = 'active'
RETURNING e.equipment_id,
          e.equipment_name,
          e.inventory_number,
          e.status AS new_status;
Применение: Логирование изменений, аудит, формирование отчёта о проведённых обновлениях.

3. DELETE — Удаление данных

Базовый синтаксис

DELETE FROM таблица
WHERE условие;
ВНИМАНИЕ: DELETE без WHERE удалит ВСЕ строки таблицы! Это необратимо без транзакции (ROLLBACK). 
-- Удалить ошибочные записи телеметрии
-- с плохим флагом качества данных
DELETE FROM fact_equipment_telemetry
WHERE quality_flag = 'ERROR'
  AND date_id = 20240315;

DELETE с подзапросом

Удаление по связанным данным

-- Удалить записи о добыче за выходные дни
-- (ошибочно загруженные данные)
DELETE FROM fact_production
WHERE date_id IN (
    SELECT date_id
    FROM dim_date
    WHERE is_weekend = TRUE
      AND year = 2024
      AND month = 3
);

-- DELETE ... USING (аналог UPDATE ... FROM в PostgreSQL)
DELETE FROM fact_equipment_telemetry t
USING dim_sensor s
WHERE t.sensor_id = s.sensor_id
  AND s.status = 'replaced'
  AND t.date_id >= 20240301;
DELETE ... USING — расширение PostgreSQL для удаления с JOIN.

DELETE ... RETURNING

Удаление с возвратом данных

-- Удалить устаревшие записи телеметрии и вернуть их
-- (для архивирования)
DELETE FROM fact_equipment_telemetry
WHERE date_id < 20230101
  AND quality_flag = 'ERROR'
RETURNING telemetry_id, date_id, sensor_id, sensor_value;
Паттерн архивации: INSERT INTO archive_table SELECT ... FROM DELETE ... RETURNING — удаление с одновременным архивированием (через CTE). 
-- Удаление + архивирование в одной транзакции (CTE)
WITH deleted AS (
    DELETE FROM fact_equipment_telemetry
    WHERE date_id < 20230101 AND quality_flag = 'ERROR'
    RETURNING *
)
INSERT INTO archive_telemetry
SELECT * FROM deleted;

4. MERGE — Слияние данных

PostgreSQL 15+ (SQL:2003 стандарт)

MERGE INTO целевая_таблица t
USING исходная_таблица s
    ON условие_соединения
WHEN MATCHED THEN
    UPDATE SET столбец = значение
WHEN NOT MATCHED THEN
    INSERT (столбцы) VALUES (значения);

MERGE = условный INSERT + UPDATE + DELETE в одном операторе

Если запись существует — обновить. Если нет — вставить. Можно также удалять.

MERGE: Синхронизация справочника

Обновление статусов оборудования из внешней системы

-- Данные от системы мониторинга приходят в staging-таблицу
MERGE INTO dim_equipment AS target
USING staging_equipment_status AS source
    ON target.inventory_number = source.inventory_number
WHEN MATCHED AND source.status IS DISTINCT FROM target.status THEN
    UPDATE SET
        status = source.status,
        has_navigation = source.has_navigation,
        has_video_recorder = source.has_video_recorder
WHEN NOT MATCHED THEN
    INSERT (equipment_type_id, mine_id, equipment_name,
            inventory_number, manufacturer, model,
            year_manufactured, status,
            has_video_recorder, has_navigation)
    VALUES (source.equipment_type_id, source.mine_id,
            source.equipment_name, source.inventory_number,
            source.manufacturer, source.model,
            source.year_manufactured, source.status,
            source.has_video_recorder, source.has_navigation);

MERGE: Три ветки

WHEN MATCHED / NOT MATCHED / NOT MATCHED BY SOURCE

-- Полная синхронизация справочника причин простоев
MERGE INTO dim_downtime_reason AS target
USING staging_downtime_reasons AS source
    ON target.reason_code = source.reason_code

WHEN MATCHED THEN
    UPDATE SET
        reason_name = source.reason_name,
        category    = source.category,
        description = source.description

WHEN NOT MATCHED THEN
    INSERT (reason_name, reason_code, category, description)
    VALUES (source.reason_name, source.reason_code,
            source.category, source.description);
Примечание: WHEN NOT MATCHED BY SOURCE с DELETE поддерживается в PostgreSQL 17+. В более ранних версиях удаление «лишних» записей выполняется отдельным DELETE.

UPSERT: INSERT ... ON CONFLICT

Альтернатива MERGE (PostgreSQL 9.5+)

-- Добавить показание датчика. Если такой уже есть — обновить значение.
INSERT INTO fact_equipment_telemetry (
    date_id, time_id, equipment_id, sensor_id,
    location_id, sensor_value, is_alarm, quality_flag
)
VALUES (
    20240315, 1430, 3, 12, 5, 87.5, FALSE, 'OK'
)
ON CONFLICT (telemetry_id) DO UPDATE SET
    sensor_value = EXCLUDED.sensor_value,
    is_alarm     = EXCLUDED.is_alarm,
    quality_flag = EXCLUDED.quality_flag,
    loaded_at    = NOW();
EXCLUDED — специальная псевдотаблица, содержащая значения, которые предполагалось вставить.

ON CONFLICT DO NOTHING

Вставить только если записи нет

-- Добавить оператора, если такой табельный номер не существует
INSERT INTO dim_operator (
    tab_number, last_name, first_name, middle_name,
    position, qualification, hire_date, mine_id
)
VALUES (
    'TAB-042', 'Козлов', 'Андрей', 'Петрович',
    'Машинист ПДМ', '5 разряд', '2025-01-15', 1
)
ON CONFLICT (tab_number) DO NOTHING;
Когда использовать: Идемпотентная загрузка справочных данных. Повторный запуск скрипта не создаст дубликатов.

5. SQL DML vs DAX

Принципиальное различие

SQL (PostgreSQL)

  • Полный DML: INSERT, UPDATE, DELETE, MERGE
  • Изменяет данные «на месте»
  • Транзакционный контроль (ACID)
  • Используется в ETL-процессах

DAX (Power BI)

  • Только чтение: DML отсутствует
  • Данные загружаются при обновлении модели
  • Power Query (M) для трансформации
  • Инкрементальное обновление

DAX — аналитический язык, не язык управления данными

В Power BI данные модифицируются через ETL-конвейер (Power Query / Dataflow), а не через запросы.

Power BI: Жизненный цикл данных

Вместо DML — обновление модели

ЭтапSQL (PostgreSQL)Power BI
Добавление данныхINSERT INTO ...Scheduled Refresh / Dataflow
Обновление данныхUPDATE ... SETFull / Incremental Refresh
Удаление данныхDELETE FROM ...Фильтрация в Power Query
Слияние данныхMERGE ... USINGMerge Queries (Power Query)
ТранзакцииBEGIN / COMMITАтомарность обновления
Ключевая разница: SQL изменяет данные «точечно» (построчно). Power BI перезагружает весь набор данных (или партицию при инкрементальном обновлении).

Power BI: Инкрементальное обновление

Аналог INSERT для больших таблиц

Сценарий «Руда+»: fact_equipment_telemetry содержит миллионы строк. Полная перезагрузка — слишком долго.
  1. Настроить Incremental Refresh в Power BI Desktop
  2. Указать параметры RangeStart и RangeEnd (дата)

6. Кейс: ETL-процесс загрузки данных MES

Сценарий «Руда+»

Задача: Каждую смену данные от датчиков оборудования, навигационной системы и операторов поступают в staging-зону. Нужно:
  1. INSERT — Загрузить новые записи о добыче
  2. UPDATE — Обновить статусы оборудования
  3. MERGE — Синхронизировать показания датчиков
  4. DELETE — Удалить ошибочные данные телеметрии

Кейс: ETL в одной транзакции

BEGIN;

-- 1. Загрузка новых записей о добыче
INSERT INTO fact_production (
    date_id, shift_id, mine_id, shaft_id,
    equipment_id, operator_id, tons_mined, trips_count
)
SELECT date_id, shift_id, mine_id, shaft_id,
       equipment_id, operator_id, tons_mined, trips_count
FROM staging_production
WHERE NOT EXISTS (
    SELECT 1 FROM fact_production fp
    WHERE fp.date_id = staging_production.date_id
      AND fp.equipment_id = staging_production.equipment_id
      AND fp.shift_id = staging_production.shift_id
);

-- 2. Обновить статусы оборудования
UPDATE dim_equipment e
SET status = s.new_status
FROM staging_equipment_status s
WHERE e.inventory_number = s.inventory_number
  AND e.status != s.new_status;

-- 3. Очистить staging
TRUNCATE staging_production, staging_equipment_status;

COMMIT;
