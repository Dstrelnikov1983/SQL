1. Функции vs Процедуры в PostgreSQL

Два механизма серверной логики

Характеристика FUNCTION PROCEDURE
Появилась в PG Изначально PostgreSQL 11+
Вызов SELECT func() CALL proc()
Возвращает значение Да (обязательно) Нет (только OUT-параметры)
В составе запроса Да (SELECT, WHERE, JOIN) Нет
Управление транзакциями Нет (COMMIT/ROLLBACK запрещены) Да (COMMIT/ROLLBACK внутри)
Основное применение Вычисления, запросы ETL, пакетная обработка

Когда что использовать?

FUNCTION

  • Нужно вернуть результат
  • Использование в SELECT / WHERE
  • Вычисляемые значения
  • Табличные отчёты
  • Агрегатные расчёты

PROCEDURE

  • ETL-процессы
  • Пакетная загрузка данных
  • Нужен COMMIT внутри
  • Обслуживание таблиц
  • Многошаговые операции
Совет «Руда+»: Для аналитических отчётов — используйте FUNCTION.
Для ежедневной загрузки данных с датчиков — используйте PROCEDURE.

2. CREATE FUNCTION — скалярный результат

Простейшая функция: расчёт KPI

-- Функция расчёта коэффициента использования оборудования (OEE)
CREATE OR REPLACE FUNCTION calc_oee(
    p_operating_hours NUMERIC,
    p_planned_hours   NUMERIC,
    p_tons_actual     NUMERIC,
    p_tons_capacity   NUMERIC,
    p_quality_rate    NUMERIC DEFAULT 0.95
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
DECLARE
    v_availability NUMERIC;
    v_performance  NUMERIC;
BEGIN
    -- Доступность = Фактическое время / Плановое время
    v_availability := p_operating_hours / NULLIF(p_planned_hours, 0);
    -- Производительность = Факт. выпуск / Макс. выпуск
    v_performance  := p_tons_actual / NULLIF(p_tons_capacity, 0);
    -- OEE = Доступность * Производительность * Качество
    RETURN ROUND(v_availability * v_performance * p_quality_rate * 100, 2);
END;
$$;
-- Вызов функции
SELECT calc_oee(10.5, 12.0, 850, 1000);
-- Результат: 83.13

Использование функции в запросе

Расчёт OEE по каждому оборудованию

SELECT
    e.equipment_name,
    m.mine_name,
    SUM(fp.operating_hours)   AS total_hours,
    SUM(fp.tons_mined)        AS total_tons,
    calc_oee(
        SUM(fp.operating_hours),
        COUNT(DISTINCT fp.shift_id) * 8.0,   -- 8 часов/смена
        SUM(fp.tons_mined),
        COUNT(DISTINCT fp.shift_id) * 120.0  -- 120 т/смена норма
    ) AS oee_percent
FROM fact_production fp
JOIN dim_equipment e ON fp.equipment_id = e.equipment_id
JOIN dim_mine m      ON fp.mine_id = m.mine_id
WHERE fp.date_id BETWEEN 20250101 AND 20250131
GROUP BY e.equipment_name, m.mine_name
ORDER BY oee_percent DESC;
Преимущество: Логика расчёта OEE определена в одном месте. Изменяя формулу в функции, мы автоматически обновляем все отчёты.

RETURNS TABLE — табличная функция

Отчёт по добыче за период

CREATE OR REPLACE FUNCTION get_production_report(
    p_date_from INT,
    p_date_to   INT,
    p_mine_id   INT DEFAULT NULL  -- NULL = все шахты
)
RETURNS TABLE (
    mine_name      VARCHAR,
    shift_name     VARCHAR,
    total_tons     NUMERIC,
    total_trips    BIGINT,
    avg_tons_trip  NUMERIC,
    equipment_cnt  BIGINT
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT
        m.mine_name,
        s.shift_name,
        ROUND(SUM(fp.tons_mined), 2),
        SUM(fp.trips_count)::BIGINT,
        ROUND(SUM(fp.tons_mined) / NULLIF(SUM(fp.trips_count), 0), 2),
        COUNT(DISTINCT fp.equipment_id)
    FROM fact_production fp
    JOIN dim_mine m  ON fp.mine_id = m.mine_id
    JOIN dim_shift s ON fp.shift_id = s.shift_id
    WHERE fp.date_id BETWEEN p_date_from AND p_date_to
      AND (p_mine_id IS NULL OR fp.mine_id = p_mine_id)
    GROUP BY m.mine_name, s.shift_name
    ORDER BY m.mine_name, s.shift_name;
END;
$$;
-- Вызов как обычную таблицу
SELECT * FROM get_production_report(20250101, 20250131);
SELECT * FROM get_production_report(20250101, 20250131, 2);

RETURNS SETOF — набор записей

Поиск аномальных показаний датчиков

CREATE OR REPLACE FUNCTION find_sensor_anomalies(
    p_date_id      INT,
    p_threshold    NUMERIC DEFAULT 2.0  -- кол-во стд. отклонений
)
RETURNS SETOF fact_equipment_telemetry
LANGUAGE plpgsql
AS $$
DECLARE
    v_avg   NUMERIC;
    v_stddev NUMERIC;
BEGIN
    -- Рассчитываем статистики за день
    SELECT AVG(sensor_value), STDDEV(sensor_value)
    INTO v_avg, v_stddev
    FROM fact_equipment_telemetry
    WHERE date_id = p_date_id;

    -- Возвращаем записи, выходящие за пределы
    RETURN QUERY
    SELECT t.*
    FROM fact_equipment_telemetry t
    WHERE t.date_id = p_date_id
      AND ABS(t.sensor_value - v_avg) > p_threshold * v_stddev;
END;
$$;
-- Вызов — возвращает все столбцы fact_equipment_telemetry
SELECT * FROM find_sensor_anomalies(20250115);
SELECT * FROM find_sensor_anomalies(20250115, 3.0);

Вопрос для обсуждения

В чём разница между RETURNS TABLE и RETURNS SETOF?


  • RETURNS SETOF тип — возвращает набор строк существующего типа (таблицы)
  • RETURNS TABLE (...) — определяет произвольную структуру результата
  • Когда что? SETOF — когда структура совпадает с таблицей; TABLE — для агрегированных/сводных отчётов

3. CREATE PROCEDURE — CALL и транзакции

Процедура загрузки сменного отчёта

CREATE OR REPLACE PROCEDURE load_daily_production(
    p_date_id INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_deleted INT;
    v_rows_inserted INT;
BEGIN
    -- Шаг 1: Удаляем старые данные за дату (перезагрузка)
    DELETE FROM fact_production WHERE date_id = p_date_id;
    GET DIAGNOSTICS v_rows_deleted = ROW_COUNT;
    RAISE NOTICE 'Удалено строк: %', v_rows_deleted;

    -- Промежуточный COMMIT — данные удалены
    COMMIT;

    -- Шаг 2: Загружаем новые данные из staging-таблицы
    INSERT INTO fact_production (
        date_id, shift_id, mine_id, shaft_id,
        equipment_id, operator_id, location_id, ore_grade_id,
        tons_mined, tons_transported, trips_count,
        distance_km, fuel_consumed_l, operating_hours
    )
    SELECT
        p_date_id, shift_id, mine_id, shaft_id,
        equipment_id, operator_id, location_id, ore_grade_id,
        tons_mined, tons_transported, trips_count,
        distance_km, fuel_consumed_l, operating_hours
    FROM staging_production
    WHERE date_id = p_date_id;

    GET DIAGNOSTICS v_rows_inserted = ROW_COUNT;
    RAISE NOTICE 'Вставлено строк: %', v_rows_inserted;

    COMMIT;
END;
$$;
-- Вызов процедуры
CALL load_daily_production(20250301);

Транзакции внутри процедур

Ключевое отличие PROCEDURE от FUNCTION

В PROCEDURE можно:

  • COMMIT — зафиксировать промежуточные результаты
  • ROLLBACK — откатить изменения до последнего COMMIT
В FUNCTION нельзя: COMMIT и ROLLBACK запрещены! Функция всегда выполняется внутри вызывающей транзакции. 
-- Пример: пакетная обработка с промежуточными COMMIT
CREATE OR REPLACE PROCEDURE batch_update_quality_flags()
LANGUAGE plpgsql
AS $$
DECLARE
    v_batch_size INT := 10000;
    v_updated INT;
BEGIN
    LOOP
        UPDATE fact_equipment_telemetry
        SET quality_flag = 'CHECKED'
        WHERE telemetry_id IN (
            SELECT telemetry_id FROM fact_equipment_telemetry
            WHERE quality_flag = 'PENDING'
            LIMIT v_batch_size
        );
        GET DIAGNOSTICS v_updated = ROW_COUNT;
        COMMIT;  -- Фиксируем каждый пакет!
        EXIT WHEN v_updated = 0;
        RAISE NOTICE 'Обработано: %', v_updated;
    END LOOP;
END;
$$;

4. Параметры: IN, OUT, INOUT, DEFAULT

Режим Описание Пример
IN Входной (по умолчанию), только для чтения p_date_id INT
OUT Выходной, записывается функцией/процедурой OUT p_count INT
INOUT Двунаправленный (вход + выход) INOUT p_value NUMERIC
DEFAULT Значение по умолчанию p_limit INT DEFAULT 100

Пример: OUT-параметры в процедуре

CREATE OR REPLACE PROCEDURE get_mine_stats(
    IN  p_mine_id       INT,
    IN  p_date_from     INT,
    IN  p_date_to       INT,
    OUT p_total_tons     NUMERIC,
    OUT p_total_downtime NUMERIC,
    OUT p_avg_oee        NUMERIC
)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Общий объём добычи
    SELECT COALESCE(SUM(tons_mined), 0)
    INTO p_total_tons
    FROM fact_production
    WHERE mine_id = p_mine_id
      AND date_id BETWEEN p_date_from AND p_date_to;

    -- Общее время простоев (в часах)
    SELECT COALESCE(SUM(duration_min) / 60.0, 0)
    INTO p_total_downtime
    FROM fact_equipment_downtime
    WHERE date_id BETWEEN p_date_from AND p_date_to
      AND equipment_id IN (
          SELECT equipment_id FROM dim_equipment
          WHERE mine_id = p_mine_id
      );

    -- Средний OEE (упрощённо)
    p_avg_oee := ROUND(p_total_tons /
        NULLIF(p_total_tons + p_total_downtime * 15, 0) * 100, 2);
END;
$$;
-- Вызов с получением OUT-параметров
CALL get_mine_stats(1, 20250101, 20250131, NULL, NULL, NULL);

DEFAULT-параметры — удобный API

CREATE OR REPLACE FUNCTION get_downtime_report(
    p_date_from    INT,
    p_date_to      INT,
    p_mine_id      INT DEFAULT NULL,        -- NULL = все шахты
    p_is_planned   BOOLEAN DEFAULT NULL,    -- NULL = все типы
    p_min_duration NUMERIC DEFAULT 0,       -- минимальная длительность
    p_limit        INT DEFAULT 100          -- кол-во записей
)
RETURNS TABLE (
    equipment_name VARCHAR,
    reason_name    VARCHAR,
    duration_hours NUMERIC,
    is_planned     BOOLEAN,
    downtime_date  DATE
)
LANGUAGE plpgsql AS $$
BEGIN
    RETURN QUERY
    SELECT
        e.equipment_name,
        dr.reason_name,
        ROUND(fd.duration_min / 60.0, 2),
        fd.is_planned,
        d.full_date
    FROM fact_equipment_downtime fd
    JOIN dim_equipment e       ON fd.equipment_id = e.equipment_id
    JOIN dim_downtime_reason dr ON fd.reason_id = dr.reason_id
    JOIN dim_date d            ON fd.date_id = d.date_id
    WHERE fd.date_id BETWEEN p_date_from AND p_date_to
      AND (p_mine_id IS NULL OR e.mine_id = p_mine_id)
      AND (p_is_planned IS NULL OR fd.is_planned = p_is_planned)
      AND fd.duration_min >= p_min_duration
    ORDER BY fd.duration_min DESC
    LIMIT p_limit;
END;
$$;
-- Различные варианты вызова
SELECT * FROM get_downtime_report(20250101, 20250131);
SELECT * FROM get_downtime_report(20250101, 20250131, p_mine_id := 2);
SELECT * FROM get_downtime_report(20250101, 20250131, p_is_planned := FALSE,
                                  p_min_duration := 60);

Циклы: LOOP, FOR, FOREACH

-- Цикл FOR по результату запроса
CREATE OR REPLACE FUNCTION generate_mine_summary(p_date_id INT)
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rec   RECORD;
    v_report TEXT := '';
BEGIN
    FOR v_rec IN
        SELECT m.mine_name,
               SUM(fp.tons_mined) AS total_tons,
               COUNT(DISTINCT fp.equipment_id) AS equip_cnt
        FROM fact_production fp
        JOIN dim_mine m ON fp.mine_id = m.mine_id
        WHERE fp.date_id = p_date_id
        GROUP BY m.mine_name
        ORDER BY total_tons DESC
    LOOP
        v_report := v_report ||
            format('Шахта: %s | Добыча: %s т | Оборудования: %s',
                   v_rec.mine_name,
                   v_rec.total_tons,
                   v_rec.equip_cnt) || E'\n';
    END LOOP;

    IF v_report = '' THEN
        v_report := 'Нет данных за указанную дату.';
    END IF;

    RETURN v_report;
END;
$$;
SELECT generate_mine_summary(20250115);

Цикл WHILE и простой LOOP

-- WHILE — расчёт нарастающего итога до целевого значения
CREATE OR REPLACE FUNCTION shifts_to_target(
    p_mine_id     INT,
    p_target_tons NUMERIC
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rec       RECORD;
    v_total     NUMERIC := 0;
    v_shifts    INT := 0;
BEGIN
    FOR v_rec IN
        SELECT tons_mined
        FROM fact_production
        WHERE mine_id = p_mine_id
        ORDER BY date_id DESC, shift_id DESC
    LOOP
        v_total := v_total + v_rec.tons_mined;
        v_shifts := v_shifts + 1;
        EXIT WHEN v_total >= p_target_tons;
    END LOOP;

    RETURN v_shifts;
END;
$$;
-- За сколько последних смен шахта 1 добыла 10 000 тонн?
SELECT shifts_to_target(1, 10000);

Мини-квиз

Что произойдёт при выполнении?

CREATE FUNCTION test_fn() RETURNS VOID
LANGUAGE plpgsql AS $$
BEGIN
    INSERT INTO dim_mine(mine_name, mine_code) VALUES('Тест', 'TST');
    COMMIT;  -- !!!
END;
$$;
A) Запись будет добавлена и зафиксирована
B) Ошибка: COMMIT недопустим в функции
C) Запись добавится, COMMIT будет проигнорирован
D) Функция не будет создана
Ответ: B — ERROR: invalid transaction termination. COMMIT/ROLLBACK запрещены в функциях. Для управления транзакциями используйте PROCEDURE.

6. Динамический SQL (Dynamic SQL)

Зачем нужен?

  • Имя таблицы/столбца неизвестно заранее
  • Запрос конструируется на основе параметров
  • Динамические сводные таблицы (pivot)
  • Универсальные отчёты по разным измерениям
-- Базовый синтаксис EXECUTE
EXECUTE 'SELECT COUNT(*) FROM ' || quote_ident(p_table_name)
INTO v_count;

-- С подстановкой параметров через format()
EXECUTE format(
    'SELECT COUNT(*) FROM %I WHERE %I = $1',
    p_table_name, p_column_name
)
INTO v_count
USING p_value;

format() — безопасное форматирование

Спецификатор Назначение Аналог
%s Строка как есть (без кавычек) —
%I Идентификатор (имя таблицы/столбца) — экранирует quote_ident()
%L Литерал (значение) — экранирует кавычки quote_literal() 
-- Примеры
SELECT format('SELECT * FROM %I WHERE %I = %L',
              'dim_mine', 'mine_name', 'Шахта №1');
-- Результат: SELECT * FROM dim_mine WHERE mine_name = 'Шахта №1'

SELECT format('CREATE INDEX %I ON %I(%I)',
              'idx_test', 'fact_production', 'date_id');
-- Результат: CREATE INDEX idx_test ON fact_production(date_id)
Правило: ВСЕГДА используйте %I для идентификаторов и %L или USING для значений. Никогда не конкатенируйте пользовательский ввод напрямую!

Пример: универсальный отчёт «Руда+»

Группировка по произвольному измерению

CREATE OR REPLACE FUNCTION production_by_dimension(
    p_dimension VARCHAR,  -- 'mine', 'shift', 'operator', 'equipment'
    p_date_from INT,
    p_date_to   INT
)
RETURNS TABLE (dimension_value VARCHAR, total_tons NUMERIC, avg_trips NUMERIC)
LANGUAGE plpgsql
AS $$
DECLARE
    v_join  TEXT;
    v_field TEXT;
BEGIN
    CASE p_dimension
        WHEN 'mine' THEN
            v_join  := 'JOIN dim_mine d ON fp.mine_id = d.mine_id';
            v_field := 'd.mine_name';
        WHEN 'shift' THEN
            v_join  := 'JOIN dim_shift d ON fp.shift_id = d.shift_id';
            v_field := 'd.shift_name';
        WHEN 'operator' THEN
            v_join  := 'JOIN dim_operator d ON fp.operator_id = d.operator_id';
            v_field := 'd.last_name || '' '' || d.first_name';
        WHEN 'equipment' THEN
            v_join  := 'JOIN dim_equipment d ON fp.equipment_id = d.equipment_id';
            v_field := 'd.equipment_name';
        ELSE
            RAISE EXCEPTION 'Неизвестное измерение: %', p_dimension;
    END CASE;

    RETURN QUERY EXECUTE format(
        'SELECT %s::VARCHAR, ROUND(SUM(fp.tons_mined),2),
                ROUND(AVG(fp.trips_count),2)
         FROM fact_production fp %s
         WHERE fp.date_id BETWEEN $1 AND $2
         GROUP BY 1 ORDER BY 2 DESC',
        v_field, v_join
    ) USING p_date_from, p_date_to;
END;
$$;
SELECT * FROM production_by_dimension('mine', 20250101, 20250131);
SELECT * FROM production_by_dimension('operator', 20250101, 20250131);

Динамическая сводная таблица (Pivot)

Добыча по шахтам — столбцы = месяцы

CREATE OR REPLACE FUNCTION pivot_production_by_month(
    p_year INT
)
RETURNS TEXT  -- возвращает SQL-запрос для выполнения
LANGUAGE plpgsql
AS $$
DECLARE
    v_sql    TEXT;
    v_cols   TEXT := '';
    v_month  INT;
BEGIN
    -- Формируем столбцы для каждого месяца
    FOR v_month IN 1..12 LOOP
        v_cols := v_cols || format(
            ', SUM(CASE WHEN d.month = %s THEN fp.tons_mined ELSE 0 END) AS %I',
            v_month,
            'month_' || LPAD(v_month::TEXT, 2, '0')
        );
    END LOOP;

    v_sql := format(
        'SELECT m.mine_name %s,
                ROUND(SUM(fp.tons_mined), 2) AS year_total
         FROM fact_production fp
         JOIN dim_mine m ON fp.mine_id = m.mine_id
         JOIN dim_date d ON fp.date_id = d.date_id
         WHERE d.year = %s
         GROUP BY m.mine_name
         ORDER BY year_total DESC',
        v_cols, p_year
    );

    RETURN v_sql;
END;
$$;
-- Получаем запрос и выполняем
SELECT pivot_production_by_month(2025);
-- Затем копируем результат и выполняем

SQL-инъекции в динамическом SQL

ОПАСНО — никогда не делайте так! 
-- УЯЗВИМЫЙ код
CREATE FUNCTION unsafe_search(p_name TEXT)
RETURNS SETOF dim_mine LANGUAGE plpgsql AS $$
BEGIN
    RETURN QUERY EXECUTE
        'SELECT * FROM dim_mine WHERE mine_name = ''' || p_name || '''';
END;
$$;

-- Атака:  p_name = "'; DROP TABLE dim_mine; --"
-- Результат: SELECT * FROM dim_mine WHERE mine_name = '';
--            DROP TABLE dim_mine; --'
БЕЗОПАСНЫЙ код: 
-- Вариант 1: USING (предпочтительно для значений)
EXECUTE 'SELECT * FROM dim_mine WHERE mine_name = $1'
USING p_name;

-- Вариант 2: format() с %L
EXECUTE format('SELECT * FROM dim_mine WHERE mine_name = %L', p_name);

-- Вариант 3: quote_literal()
EXECUTE 'SELECT * FROM dim_mine WHERE mine_name = '
    || quote_literal(p_name);

Безопасность: SECURITY DEFINER vs INVOKER

SECURITY INVOKER (по умолчанию)

Функция выполняется с правами вызывающего пользователя

  • Безопаснее
  • Пользователю нужны права на все таблицы

SECURITY DEFINER

Функция выполняется с правами создателя (владельца)

  • Контролируемый доступ
  • Пользователю нужны права только на функцию
-- Безопасный доступ к отчёту без доступа к таблицам
CREATE OR REPLACE FUNCTION public_production_report(p_date_id INT)
RETURNS TABLE (mine VARCHAR, tons NUMERIC)
LANGUAGE plpgsql
SECURITY DEFINER  -- выполняется от имени владельца
SET search_path = public  -- Важно! Фиксируем search_path
AS $$
BEGIN
    RETURN QUERY
    SELECT m.mine_name, SUM(fp.tons_mined)
    FROM fact_production fp
    JOIN dim_mine m ON fp.mine_id = m.mine_id
    WHERE fp.date_id = p_date_id
    GROUP BY m.mine_name;
END;
$$;

-- Даём права аналитикам только на функцию
GRANT EXECUTE ON FUNCTION public_production_report(INT) TO analyst_role;

7. Сравнение с DAX

Повторно используемая логика

PostgreSQL

-- Хранимая функция
CREATE FUNCTION get_oee(
    p_mine_id INT,
    p_date_id INT
)
RETURNS NUMERIC AS $$
  SELECT ROUND(
    SUM(fp.operating_hours) /
    NULLIF(
      COUNT(DISTINCT fp.shift_id)
      * 8.0, 0
    ) * 100, 2
  )
  FROM fact_production fp
  WHERE fp.mine_id = p_mine_id
    AND fp.date_id = p_date_id;
$$ LANGUAGE sql;

DAX (мера в Power BI)

// Мера — аналог функции
OEE =
VAR _OperHours =
    SUM(fact_production[operating_hours])
VAR _PlannedHours =
    DISTINCTCOUNT(
      fact_production[shift_id]
    ) * 8
RETURN
    DIVIDE(
        _OperHours,
        _PlannedHours,
        0
    ) * 100
Ключевое различие: В DAX нет процедур. Мера (measure) — ближайший аналог функции, но она всегда вычисляется в контексте фильтра.

DAX: вычисляемая таблица vs табличная функция

PostgreSQL — RETURNS TABLE

    20250101, 20250131, 2
);

-- Параметры передаются явно
-- Результат — набор строк
-- Можно использовать в JOIN

DAX — вычисляемая таблица

// Вычисляемая таблица
ProductionReport =
ADDCOLUMNS(
    SUMMARIZE(
        fact_production,
        dim_mine[mine_name],
        dim_shift[shift_name]

    "TotalTons",
        SUM(fact_production[tons_mined]),
    "AvgTrips",
        AVERAGE(fact_production[trips_count])
)
// Параметры = контекст фильтра
// (слайсеры в отчёте)

Что есть в SQL, но нет в DAX

Возможность PostgreSQL DAX
Хранимые процедуры CREATE PROCEDURE + CALL Нет
Управление транзакциями COMMIT / ROLLBACK Нет
Динамический SQL EXECUTE + format() Нет
Циклы и условия IF, FOR, LOOP Ограниченно (IF, SWITCH)
Параметризованные запросы IN/OUT параметры Через What-If параметры
Повторно используемая логика Функции Меры и вычисляемые таблицы
Модификация данных INSERT/UPDATE/DELETE Только чтение
Вывод: SQL-функции и процедуры — серверная логика для ETL и аналитики. DAX-меры — клиентская логика для интерактивных отчётов.

Кейс для анализа

Проблема производительности на «Руда+»

Аналитик написал функцию с динамическим SQL, которая принимает произвольное имя таблицы и столбца от пользователя:

CREATE FUNCTION bad_report(p_table TEXT, p_column TEXT, p_value TEXT)
RETURNS INT LANGUAGE plpgsql AS $$
DECLARE v_cnt INT;
BEGIN
    EXECUTE 'SELECT COUNT(*) FROM ' || p_table
         || ' WHERE ' || p_column || ' = ''' || p_value || ''''
    INTO v_cnt;
    RETURN v_cnt;
END;
$$;
Найдите все проблемы:
  1. SQL-инъекция через p_table — нет quote_ident()
  2. SQL-инъекция через p_value — нет quote_literal()
  3. Нет валидации имени таблицы (можно обратиться к системным таблицам)
  4. Нет SECURITY DEFINER / ограничения прав

Исправленная версия

CREATE FUNCTION safe_report(
    p_table  TEXT,
    p_column TEXT,
    p_value  TEXT
)
RETURNS INT
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
    v_cnt INT;
    v_allowed_tables TEXT[] := ARRAY[
        'fact_production', 'fact_equipment_downtime',
        'fact_ore_quality', 'fact_equipment_telemetry'
    ];
BEGIN
    -- Валидация: только разрешённые таблицы
    IF NOT (p_table = ANY(v_allowed_tables)) THEN
        RAISE EXCEPTION 'Таблица "%" не разрешена', p_table;
    END IF;

    -- Безопасный динамический SQL
    EXECUTE format(
        'SELECT COUNT(*) FROM %I WHERE %I = $1',
        p_table, p_column
    )
    INTO v_cnt
    USING p_value;

    RETURN v_cnt;
END;
$$;
Ключевые улучшения: белый список таблиц, format(%I) для идентификаторов, USING для значений, SECURITY DEFINER + фиксация search_path.

Ответ: C — %I (Identifier) экранирует имена таблиц/столбцов. %L — для литеральных значений.

Практические рекомендации

Именование

  • Функции: get_*, calc_*, find_*, check_*
  • Процедуры: load_*, update_*, process_*, sync_*

Производительность

  • Используйте LANGUAGE sql для простых функций (инлайнятся)
  • STABLE / IMMUTABLE — позволяет оптимизатору кэшировать
  • Избегайте VOLATILE функций в WHERE — планировщик не может оптимизировать

Безопасность

  • Всегда SET search_path при SECURITY DEFINER
  • Минимальные привилегии: GRANT EXECUTE только нужным ролям
  • Валидация входных параметров перед динамическим SQL

Категория Описание Пример
