1. Анонимные блоки DO $$ ... END $$

Что такое анонимный блок?

Фрагмент PL/pgSQL-кода, который выполняется немедленно, не сохраняется в БД и не имеет имени.

DO $$
DECLARE
    -- Объявление переменных
BEGIN
    -- Тело блока: SQL-операторы, логика
EXCEPTION
    -- Обработка ошибок (необязательно)
END;
$$;
Когда использовать: одноразовые скрипты, миграции данных, быстрые проверки, прототипирование логики перед созданием функции.

Анонимный блок: пример «Руда+»

DO $$
DECLARE
    v_total_mines   INT;
    v_total_equip   INT;
    v_total_sensors INT;
BEGIN
    SELECT COUNT(*) INTO v_total_mines   FROM dim_mine;
    SELECT COUNT(*) INTO v_total_equip   FROM dim_equipment;
    SELECT COUNT(*) INTO v_total_sensors FROM dim_sensor;

    RAISE NOTICE 'Статистика «Руда+»:';
    RAISE NOTICE '  Шахт: %', v_total_mines;
    RAISE NOTICE '  Оборудования: %', v_total_equip;
    RAISE NOTICE '  Датчиков: %', v_total_sensors;
END;
$$;
Результат: В панели Messages появятся строки с уведомлениями. Анонимные блоки не возвращают результат в виде таблицы — только сообщения.

2. Переменные: DECLARE

DO $$
DECLARE
    -- Простые типы с инициализацией
    v_mine_name    VARCHAR(100) := 'Шахта Северная';
    v_shift_count  INT          := 3;
    v_target_tons  NUMERIC(10,2) DEFAULT 500.00;
    v_is_active    BOOLEAN      := TRUE;
    v_report_date  DATE         := CURRENT_DATE;

    -- Тип на основе столбца таблицы
    v_equip_name   dim_equipment.equipment_name%TYPE;

    -- Вся строка таблицы
    v_mine_row     dim_mine%ROWTYPE;

    -- Константа
    c_max_fe       CONSTANT NUMERIC := 72.0;

    -- Без инициализации (будет NULL)
    v_result       TEXT;
BEGIN
    -- ...
END;
$$;

Способы присваивания

Способ Синтаксис Когда использовать
При объявлении v_name VARCHAR := 'значение'; Начальное значение
DEFAULT v_name VARCHAR DEFAULT 'значение'; Альтернатива :=
Оператор := v_name := 'новое значение'; В теле блока
SELECT INTO SELECT col INTO v_name FROM ... Результат запроса 
DO $$
DECLARE
    v_mine_name  VARCHAR;
    v_total_tons NUMERIC;
BEGIN
    -- SELECT INTO: результат запроса -> переменная
    SELECT m.mine_name, SUM(fp.tons_mined)
    INTO v_mine_name, v_total_tons
    FROM fact_production fp
    JOIN dim_mine m ON fp.mine_id = m.mine_id
    WHERE fp.mine_id = 1
      AND fp.date_id BETWEEN 20250101 AND 20250131
    GROUP BY m.mine_name;

    RAISE NOTICE '% — добыто: % т', v_mine_name, v_total_tons;
END;
$$;

%TYPE и %ROWTYPE

%TYPE

Наследует тип одного столбца

DECLARE
  v_name dim_mine.mine_name%TYPE;
  -- Эквивалент: VARCHAR(100)
  -- Но автоматически обновится,
  -- если тип столбца изменится

%ROWTYPE

Наследует структуру всей строки

DECLARE
  v_row dim_equipment%ROWTYPE;
BEGIN
  SELECT * INTO v_row
  FROM dim_equipment
  WHERE equipment_id = 1;
  -- v_row.equipment_name
  -- v_row.mine_id и т.д.
Рекомендация: Используйте %TYPE — если столбец изменит тип (например, с VARCHAR(100) на VARCHAR(200)), ваш код автоматически адаптируется.

3. Условные конструкции: IF

DO $$
DECLARE
    v_fe_content NUMERIC := 58.5;
    v_grade      VARCHAR;
BEGIN
    IF v_fe_content >= 65 THEN
        v_grade := 'Высший сорт (богатая руда)';
    ELSIF v_fe_content >= 55 THEN
        v_grade := 'Первый сорт';
    ELSIF v_fe_content >= 45 THEN
        v_grade := 'Второй сорт';
    ELSIF v_fe_content >= 30 THEN
        v_grade := 'Третий сорт (бедная руда)';
    ELSE
        v_grade := 'Пустая порода';
    END IF;

    RAISE NOTICE 'Fe: %, Сорт: %', v_fe_content, v_grade;
END;
$$;
Обратите внимание: В PL/pgSQL используется ELSIF (не ELSEIF и не ELSE IF). Каждый IF должен заканчиваться END IF;.

Оператор CASE в PL/pgSQL

Простой CASE

CASE v_equipment_type
    WHEN 'ПДМ' THEN
        v_role := 'Погрузка';
    WHEN 'Самосвал' THEN
        v_role := 'Транспорт';
    WHEN 'Вагонетка' THEN
        v_role := 'Откатка';
    ELSE
        v_role := 'Прочее';
END CASE;

Поисковый CASE

CASE
    WHEN v_hours > 100 THEN
        v_severity := 'КРИТИЧНО';
    WHEN v_hours > 50 THEN
        v_severity := 'ВЫСОКАЯ';
    WHEN v_hours > 20 THEN
        v_severity := 'СРЕДНЯЯ';
    ELSE
        v_severity := 'НИЗКАЯ';
END CASE;
Важно: CASE в PL/pgSQL — это оператор (заканчивается END CASE;). Не путайте с выражением CASE в SQL (заканчивается END).

Пример: оценка эффективности смены

DO $$
DECLARE
    rec RECORD;
    v_status VARCHAR;
    v_target NUMERIC := 5000;  -- Целевая добыча на шахту за месяц
BEGIN
    FOR rec IN
        SELECT m.mine_name, COALESCE(SUM(fp.tons_mined), 0) AS total_tons
        FROM dim_mine m
        LEFT JOIN fact_production fp ON fp.mine_id = m.mine_id
            AND fp.date_id BETWEEN 20250101 AND 20250131
        GROUP BY m.mine_name
        ORDER BY total_tons DESC
    LOOP
        IF rec.total_tons >= v_target THEN
            v_status := 'ПЛАН ВЫПОЛНЕН';
        ELSIF rec.total_tons >= v_target * 0.8 THEN
            v_status := 'БЛИЗКО К ПЛАНУ';
        ELSIF rec.total_tons > 0 THEN
            v_status := 'ОТСТАВАНИЕ';
        ELSE
            v_status := 'НЕТ ДАННЫХ';
        END IF;

        RAISE NOTICE '% — % т [%]', rec.mine_name, rec.total_tons, v_status;
    END LOOP;
END;
$$;

Вопрос для обсуждения

Сравните два подхода к классификации:

CASE в SQL

SELECT mine_name,
    CASE
        WHEN total > 5000 THEN 'OK'
        WHEN total > 3000 THEN 'WARN'
        ELSE 'CRIT'
    END AS status
FROM ...

IF в PL/pgSQL

IF total > 5000 THEN
    status := 'OK';
ELSIF total > 3000 THEN
    status := 'WARN';
ELSE
    status := 'CRIT';
END IF;
Вопросы:
  • Когда достаточно CASE в SQL, а когда нужен IF в PL/pgSQL?
  • Какой подход быстрее? Почему?

4. Циклы в PL/pgSQL

Цикл Назначение Пример
LOOP Бесконечный цикл (нужен EXIT) Ожидание условия
WHILE Пока условие истинно Накопление до порога
FOR i IN 1..N Числовой диапазон Помесячный отчёт
FOR rec IN query Итерация по запросу Обработка строк
FOREACH x IN ARRAY Итерация по массиву Список шахт
Управление циклом: EXIT — выход из цикла, CONTINUE — переход к следующей итерации. Оба поддерживают WHEN условие.

LOOP с EXIT и CONTINUE

DO $$
DECLARE
    v_counter INT := 0;
    v_total   NUMERIC := 0;
    v_daily   NUMERIC;
BEGIN
    LOOP
        v_counter := v_counter + 1;

        -- Выход по условию
        EXIT WHEN v_counter > 10;

        SELECT COALESCE(SUM(tons_mined), 0)
        INTO v_daily
        FROM fact_production
        WHERE date_id = 20250100 + v_counter;

        -- Пропуск дней без добычи
        CONTINUE WHEN v_daily = 0;

        v_total := v_total + v_daily;
        RAISE NOTICE 'День %: % т (итого: % т)',
            v_counter, v_daily, v_total;
    END LOOP;

    RAISE NOTICE 'Итого за 10 дней: % т', v_total;
END;
$$;

WHILE — ожидание выполнения плана

DO $$
DECLARE
    v_date_id       INT := 20250101;
    v_accumulated   NUMERIC := 0;
    v_monthly_target NUMERIC := 50000;
    v_daily_prod    NUMERIC;
BEGIN
    WHILE v_accumulated < v_monthly_target AND v_date_id <= 20250131
    LOOP
        SELECT COALESCE(SUM(tons_mined), 0)
        INTO v_daily_prod
        FROM fact_production
        WHERE date_id = v_date_id;

        v_accumulated := v_accumulated + v_daily_prod;
        v_date_id := v_date_id + 1;
    END LOOP;

    IF v_accumulated >= v_monthly_target THEN
        RAISE NOTICE 'План % т выполнен к дате %',
            v_monthly_target, v_date_id - 1;
    ELSE
        RAISE NOTICE 'План НЕ выполнен: % т из % т',
            v_accumulated, v_monthly_target;
    END IF;
END;
$$;

FOR — числовой диапазон и запрос

Числовой FOR

-- Помесячный отчёт
FOR v_month IN 1..12 LOOP
    SELECT SUM(tons_mined)
    INTO v_prod
    FROM fact_production
    WHERE date_id BETWEEN
      (20250000 + v_month*100 + 1) AND
      (20250000 + v_month*100 + 31);

    RAISE NOTICE 'Месяц %: % т',
        v_month, v_prod;
END LOOP;

-- Обратный порядок
FOR i IN REVERSE 10..1 LOOP
    RAISE NOTICE 'Обратный: %', i;
END LOOP;

FOR по запросу

-- ТОП-5 шахт
FOR rec IN
    SELECT m.mine_name,
           SUM(fp.tons_mined) AS tons
    FROM fact_production fp
    JOIN dim_mine m
      ON fp.mine_id = m.mine_id
    WHERE fp.date_id BETWEEN
      20250101 AND 20250131
    GROUP BY m.mine_name
    ORDER BY tons DESC
    LIMIT 5
LOOP
    RAISE NOTICE '% — % т',
        rec.mine_name, rec.tons;
END LOOP;

FOREACH — итерация по массиву

DO $$
DECLARE
    v_mine_ids INT[] := ARRAY[1, 2, 3, 4, 5];
    v_mine_id  INT;
    v_name     VARCHAR;
    v_tons     NUMERIC;
BEGIN
    FOREACH v_mine_id IN ARRAY v_mine_ids
    LOOP
        SELECT m.mine_name, COALESCE(SUM(fp.tons_mined), 0)
        INTO v_name, v_tons
        FROM dim_mine m
        LEFT JOIN fact_production fp ON fp.mine_id = m.mine_id
            AND fp.date_id BETWEEN 20250101 AND 20250131
        WHERE m.mine_id = v_mine_id
        GROUP BY m.mine_name;

        RAISE NOTICE 'Шахта «%» (ID=%): % т', v_name, v_mine_id, v_tons;
    END LOOP;
END;
$$;
Когда FOREACH? Когда набор значений задан явно (параметр-массив, конфигурация), а не получен из запроса.

Форматирование RAISE

DO $$
DECLARE
    v_mine     VARCHAR := 'Северная';
    v_tons     NUMERIC := 1234.56;
    v_target   NUMERIC := 1500.00;
    v_percent  NUMERIC;
BEGIN
    v_percent := ROUND(v_tons / v_target * 100, 1);

    -- % — подставляет значение аргумента
    RAISE NOTICE 'Шахта «%»: добыто % т из % т (% %%)',
        v_mine, v_tons, v_target, v_percent;
    -- %% — экранированный знак процента

    -- USING для дополнительной информации
    IF v_percent < 80 THEN
        RAISE WARNING 'Отставание от плана на шахте «%»!', v_mine
            USING DETAIL = format('Факт: %s т, План: %s т', v_tons, v_target),
                  HINT   = 'Проверьте журнал простоев';
    END IF;
END;
$$;
Параметры USING: DETAIL — детали, HINT — рекомендация, ERRCODE — код ошибки SQLSTATE.

6. Возврат данных из функций

Оператор Назначение Тип функции
RETURN expr Вернуть одно значение RETURNS скаляр
RETURN NEXT Добавить строку к результату RETURNS TABLE / SETOF
RETURN QUERY Добавить результат запроса RETURNS TABLE / SETOF 
-- RETURN: скалярная функция
CREATE FUNCTION get_total_production(p_mine_id INT, p_date INT)
RETURNS NUMERIC LANGUAGE plpgsql STABLE AS $$
DECLARE v_total NUMERIC;
BEGIN
    SELECT COALESCE(SUM(tons_mined), 0) INTO v_total
    FROM fact_production
    WHERE mine_id = p_mine_id AND date_id = p_date;
    RETURN v_total;
END;
$$;

RETURN NEXT — построчный возврат

CREATE OR REPLACE FUNCTION get_monthly_production_report(p_year INT)
RETURNS TABLE (
    month_num INT, month_name VARCHAR,
    total_tons NUMERIC, avg_per_day NUMERIC
) LANGUAGE plpgsql STABLE AS $$
DECLARE
    v_months VARCHAR[] := ARRAY[
        'Январь','Февраль','Март','Апрель','Май','Июнь',
        'Июль','Август','Сентябрь','Октябрь','Ноябрь','Декабрь'];
    v_days INT;
BEGIN
    FOR i IN 1..12 LOOP
        month_num := i;
        month_name := v_months[i];

        SELECT COALESCE(SUM(fp.tons_mined), 0)
        INTO total_tons
        FROM fact_production fp
        WHERE fp.date_id BETWEEN (p_year*10000 + i*100 + 1)
                              AND (p_year*10000 + i*100 + 31);

        SELECT COUNT(DISTINCT date_id) INTO v_days
        FROM fact_production
        WHERE date_id BETWEEN (p_year*10000 + i*100 + 1)
                          AND (p_year*10000 + i*100 + 31);

        avg_per_day := CASE WHEN v_days > 0
            THEN ROUND(total_tons / v_days, 1) ELSE 0 END;

        RETURN NEXT;  -- Добавляем строку к результату
    END LOOP;
END;
$$;

SELECT * FROM get_monthly_production_report(2025);

RETURN QUERY — возврат целого запроса

CREATE OR REPLACE FUNCTION get_equipment_downtime_summary(
    p_mine_id INT, p_date_from INT, p_date_to INT
)
RETURNS TABLE (
    equipment_name VARCHAR, downtime_hours NUMERIC,
    incident_count BIGINT, top_reason VARCHAR
) LANGUAGE plpgsql STABLE AS $$
BEGIN
    RETURN QUERY
    SELECT
        e.equipment_name,
        ROUND(SUM(fd.downtime_minutes) / 60.0, 1),
        COUNT(*),
        (SELECT dr.reason_name
         FROM fact_equipment_downtime fd2
         JOIN dim_downtime_reason dr ON fd2.reason_id = dr.reason_id
         WHERE fd2.equipment_id = e.equipment_id
           AND fd2.date_id BETWEEN p_date_from AND p_date_to
         GROUP BY dr.reason_name ORDER BY COUNT(*) DESC LIMIT 1)
    FROM fact_equipment_downtime fd
    JOIN dim_equipment e ON fd.equipment_id = e.equipment_id
    WHERE fd.date_id BETWEEN p_date_from AND p_date_to
      AND e.mine_id = p_mine_id
    GROUP BY e.equipment_id, e.equipment_name
    ORDER BY SUM(fd.downtime_minutes) DESC;
END;
$$;
RETURN QUERY проще, когда результат можно получить одним SQL. RETURN NEXT — когда нужна дополнительная логика (циклы, условия, вычисления).

7. Курсоры

Что такое курсор?

Указатель на результат запроса, позволяющий обрабатывать строки по одной, не загружая весь результат в память.

Операция Описание
DECLARE cur CURSOR FOR ... Объявление курсора
OPEN cur Открытие (выполнение запроса)
FETCH cur INTO rec Получение следующей строки
CLOSE cur Закрытие курсора
FOR rec IN cur LOOP Автоматический open/fetch/close

Курсор: ручное управление

DO $$
DECLARE
    cur_equipment CURSOR FOR
        SELECT e.equipment_id, e.equipment_name, et.equipment_type_name
        FROM dim_equipment e
        JOIN dim_equipment_type et ON e.equipment_type_id = et.equipment_type_id
        ORDER BY e.equipment_id;
    rec RECORD;
    v_count INT := 0;
BEGIN
    OPEN cur_equipment;

    LOOP
        FETCH cur_equipment INTO rec;
        EXIT WHEN NOT FOUND;

        v_count := v_count + 1;
        RAISE NOTICE '#%: % [%]',
            v_count, rec.equipment_name, rec.equipment_type_name;
    END LOOP;

    CLOSE cur_equipment;
    RAISE NOTICE 'Всего: %', v_count;
END;
$$;

Курсор с параметрами

DO $$
DECLARE
    cur_production CURSOR (p_mine_id INT, p_date INT) FOR
        SELECT e.equipment_name, fp.tons_mined, fp.trips_count
        FROM fact_production fp
        JOIN dim_equipment e ON fp.equipment_id = e.equipment_id
        WHERE fp.mine_id = p_mine_id AND fp.date_id = p_date
        ORDER BY fp.tons_mined DESC;
    rec RECORD;
BEGIN
    RAISE NOTICE '=== Шахта 1, 15.01.2025 ===';

    -- Открываем с параметрами
    OPEN cur_production(1, 20250115);
    LOOP
        FETCH cur_production INTO rec;
        EXIT WHEN NOT FOUND;
        RAISE NOTICE '  % — % т, % рейсов',
            rec.equipment_name, rec.tons_mined, rec.trips_count;
    END LOOP;
    CLOSE cur_production;
END;
$$;

Курсор: пакетная обработка данных

DO $$
DECLARE
    cur_dates CURSOR FOR
        SELECT DISTINCT date_id FROM fact_production
        WHERE date_id BETWEEN 20250101 AND 20250107
        ORDER BY date_id;
    v_rows INT;
    v_total_rows INT := 0;
BEGIN
    FOR rec IN cur_dates LOOP
        INSERT INTO report_daily_production (
            report_date, mine_id, mine_name,
            total_tons, total_trips, avg_tons_per_trip
        )
        SELECT d.full_date, m.mine_id, m.mine_name,
            SUM(fp.tons_mined), SUM(fp.trips_count),
            ROUND(SUM(fp.tons_mined) / NULLIF(SUM(fp.trips_count), 0), 2)
        FROM fact_production fp
        JOIN dim_mine m ON fp.mine_id = m.mine_id
        JOIN dim_date d ON fp.date_id = d.date_id
        WHERE fp.date_id = rec.date_id
        GROUP BY d.full_date, m.mine_id, m.mine_name;

        GET DIAGNOSTICS v_rows = ROW_COUNT;
        v_total_rows := v_total_rows + v_rows;
        RAISE NOTICE 'Дата %: вставлено % записей', rec.date_id, v_rows;
    END LOOP;

    RAISE NOTICE 'Всего: %', v_total_rows;
END;
$$;
FOR rec IN cursor — автоматически открывает, извлекает и закрывает курсор. Рекомендуемый способ!

    "Третий"
)

Что есть в PL/pgSQL, но нет в DAX

Возможность PL/pgSQL DAX
Анонимные блоки DO $$ ... END $$ Нет
Переменные DECLARE, :=, SELECT INTO VAR (только в мерах)
Циклы FOR, WHILE, LOOP, FOREACH Нет (только итераторы: SUMX, FILTER)
Условия IF/ELSIF/ELSE, CASE IF, SWITCH (только выражения)
Курсоры CURSOR, FETCH Нет
Вывод сообщений RAISE NOTICE/WARNING Нет
Модификация данных INSERT/UPDATE/DELETE Только чтение
Вывод: DAX — декларативный язык для аналитики. PL/pgSQL — императивный язык для серверной логики, ETL и управления данными.

Квиз: проверка знаний

1. Как объявить переменную, тип которой совпадает с типом столбца таблицы?

A) v_name := dim_mine.mine_name;
B) v_name dim_mine.mine_name%TYPE;
C) v_name TYPE OF dim_mine.mine_name;
D) v_name LIKE dim_mine.mine_name;
Ответ: B — %TYPE наследует тип столбца. Если тип столбца изменится, переменная автоматически адаптируется.

Квиз (продолжение)

2. Какой уровень RAISE прерывает выполнение блока?

A) RAISE NOTICE
B) RAISE WARNING
C) RAISE LOG
D) RAISE EXCEPTION
Ответ: D — Только RAISE EXCEPTION прерывает выполнение и вызывает откат транзакции (если нет блока EXCEPTION).

Квиз (продолжение)

3. Чем отличается RETURN NEXT от RETURN QUERY?

A) RETURN NEXT быстрее
B) RETURN NEXT добавляет одну строку, RETURN QUERY — результат запроса
C) RETURN QUERY нельзя использовать в циклах
D) Они идентичны
Ответ: B — RETURN NEXT добавляет текущие значения OUT-переменных как одну строку. RETURN QUERY добавляет сразу все строки результата запроса.

Квиз (продолжение)

4. Какой цикл подходит для перебора элементов массива?

A) FOR i IN ARRAY
B) WHILE ARRAY_LENGTH > 0
C) FOREACH x IN ARRAY
D) LOOP OVER ARRAY
Ответ: C — FOREACH x IN ARRAY arr LOOP — специальная конструкция для перебора массивов в PL/pgSQL.

Кейс для анализа

Проблема на «Руда+»: бесконечный цикл

Инженер написал скрипт для обхода всех дат, но он «зависает»:

DO $$
DECLARE
    v_date_id INT := 20250101;
    v_prod    NUMERIC;
BEGIN
    LOOP
        SELECT SUM(tons_mined) INTO v_prod
        FROM fact_production WHERE date_id = v_date_id;

        RAISE NOTICE 'Дата %: % т', v_date_id, v_prod;
        v_date_id := v_date_id + 1;
    END LOOP;
END;
$$;
Найдите проблему:

Нет условия EXIT WHEN — цикл никогда не завершится! Нужно добавить: EXIT WHEN v_date_id > 20250131;
